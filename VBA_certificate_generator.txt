Option Explicit

Sub GenerateCertificates()

    Dim WordApp As Object
    Dim WordDoc As Object

    Dim ws As Worksheet
    Dim LastRow As Long
    Dim i As Long

    Dim TemplatePath As String
    Dim OutputFolder As String

    Dim fd As FileDialog

    Set ws = ThisWorkbook.Sheets("list")

    '=============================
    'Select Word Template
    '=============================
    Set fd = Application.FileDialog(msoFileDialogFilePicker)

    With fd
        .Title = "Select Certificate Template"
        .Filters.Clear
        .Filters.Add "Word Files", "*.docx"

        If .Show <> -1 Then Exit Sub

        TemplatePath = .SelectedItems(1)
    End With

    '=============================
    'Select Output Folder
    '=============================
    Set fd = Application.FileDialog(msoFileDialogFolderPicker)

    With fd
        .Title = "Select Output Folder"

        If .Show <> -1 Then Exit Sub

        OutputFolder = .SelectedItems(1)
    End With

    '=============================
    'Open Word
    '=============================
    Set WordApp = CreateObject("Word.Application")
    WordApp.Visible = False

    LastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

    For i = 2 To LastRow

        Set WordDoc = WordApp.Documents.Open(TemplatePath)

        ReplacePlaceholder WordDoc, "<<NAME>>", ws.Cells(i, 1).Value
        ReplacePlaceholder WordDoc, "<<FROM_DATE>>", Format(ws.Cells(i, 2).Value, "dd mmmm yyyy")
        ReplacePlaceholder WordDoc, "<<TO_DATE>>", Format(ws.Cells(i, 3).Value, "dd mmmm yyyy")
        ReplacePlaceholder WordDoc, "<<CERTIFICATE_NO>>", ws.Cells(i, 4).Value

        WordDoc.ExportAsFixedFormat _
            OutputFolder & "\" & ws.Cells(i, 4).Value & "_" & ws.Cells(i, 1).Value & ".pdf", _
            17

        WordDoc.Close False

    Next i

    WordApp.Quit

    Set WordDoc = Nothing
    Set WordApp = Nothing

    MsgBox "Certificates Generated Successfully!", vbInformation

End Sub


Sub ReplacePlaceholder(ByVal Doc As Object, _
                       ByVal FindText As String, _
                       ByVal ReplaceWith As String)

    With Doc.Content.Find

        .ClearFormatting
        .Replacement.ClearFormatting

        .Text = FindText
        .Replacement.Text = ReplaceWith

        .Forward = True
        .Wrap = 1
        .Format = False
        .MatchCase = False
        .MatchWholeWord = False

        .Execute Replace:=2

    End With

End Sub
